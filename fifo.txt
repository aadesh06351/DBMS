#include <stdio.h>

int main() {
    int bt[5], wt[5], n, i;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    for(i=0;i<n;i++)
        scanf("%d",&bt[i]);

    wt[0]=0;

    for(i=1;i<n;i++)
        wt[i]=wt[i-1]+bt[i-1];

    for(i=0;i<n;i++)
        printf("P%d WT=%d\n",i+1,wt[i]);

    return 0;
}

int main() {
    int bt[5], n, i, j, temp;

    printf("Enter number of processes: ");
    scanf("%d",&n);

    for(i=0;i<n;i++)
        scanf("%d",&bt[i]);

    for(i=0;i<n;i++) {
        for(j=i+1;j<n;j++) {
            if(bt[i]>bt[j]) {
                temp=bt[i];
                bt[i]=bt[j];
                bt[j]=temp;
            }
        }
    }

    printf("Execution Order:\n");

    for(i=0;i<n;i++)
        printf("%d ",bt[i]);

    return 0;
}

#include <stdio.h>

int main() {
    int bt[5], rem[5], n, tq, i, time=0;

    printf("Enter number of processes: ");
    scanf("%d",&n);

    for(i=0;i<n;i++) {
        scanf("%d",&bt[i]);
        rem[i]=bt[i];
    }

    printf("Enter Time Quantum: ");
    scanf("%d",&tq);

    while(1) {
        int done=1;

        for(i=0;i<n;i++) {
            if(rem[i]>0) {
                done=0;

                if(rem[i]>tq) {
                    time+=tq;
                    rem[i]-=tq;
                } else {
                    time+=rem[i];
                    printf("P%d completed at %d\n",i+1,time);
                    rem[i]=0;
                }
            }
        }

        if(done)
            break;
    }

    return 0;
}
