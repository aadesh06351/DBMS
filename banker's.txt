int main() {
    int bt[10], wt[10], n, i;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    for(i = 0; i < n; i++) {
        printf("Burst time P%d: ", i+1);
        scanf("%d", &bt[i]);
    }

    wt[0] = 0;

    for(i = 1; i < n; i++)
        wt[i] = wt[i-1] + bt[i-1];

    for(i = 0; i < n; i++)
        printf("P%d Waiting Time = %d\n", i+1, wt[i]);

    return 0;
}

#include <stdio.h>

int main() {
    int frames[3], pages[10], i, j, n, fault = 0;

    printf("Enter number of pages: ");
    scanf("%d", &n);

    printf("Enter page numbers: ");
    for(i = 0; i < n; i++)
        scanf("%d", &pages[i]);

    for(i = 0; i < 3; i++)
        frames[i] = -1;

    for(i = 0; i < n; i++) {
        int found = 0;

        for(j = 0; j < 3; j++) {
            if(frames[j] == pages[i]) {
                found = 1;
                break;
            }
        }

        if(!found) {
            frames[i % 3] = pages[i];
            fault++;
        }
    }

    printf("Page Faults = %d", fault);

    return 0;
}

#include <stdio.h>

int main() {
    int block[5] = {100, 500, 200, 300, 600};
    int process[4] = {212, 417, 112, 426};
    int i, j;

    for(i = 0; i < 4; i++) {
        for(j = 0; j < 5; j++) {
            if(block[j] >= process[i]) {
                printf("Process %d allocated to block %d\n", i+1, j+1);
                block[j] -= process[i];
                break;
            }
        }
    }

    return 0;
}

#include <stdio.h>

int main() {
    int block[5] = {100, 500, 200, 300, 600};
    int process[4] = {212, 417, 112, 426};
    int i, j, best;

    for(i = 0; i < 4; i++) {
        best = -1;

        for(j = 0; j < 5; j++) {
            if(block[j] >= process[i]) {
                if(best == -1 || block[j] < block[best])
                    best = j;
            }
        }

        if(best != -1) {
            printf("Process %d allocated to block %d\n", i+1, best+1);
            block[best] -= process[i];
        }
    }

    return 0;
}
